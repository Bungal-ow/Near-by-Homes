This is the schema for cassandra

CREATE KEYSPACE bungal-ow
WITH replication = {'class':'SimpleStrategy', 'replication_factor' : 3};;

CREATE TABLE bungal-ow.houses (
    neighborhood VARCHAR,
    houseId uuid,
    transit_score INT,
    walk_score INT,
    value_inc_dec_past INT,
    value_inc_dec_future INT,
    median_value INT,
    home_cost INT ,
    bedrooms INT,
    bathrooms INT,
    home_address VARCHAR,
    sf INT,
    home_image VARCHAR
    PRIMARY KEY ((neighborhood, houseId), (home_cost)) WITH CLUSTERIMG ORDER BY (home_cost)
)
